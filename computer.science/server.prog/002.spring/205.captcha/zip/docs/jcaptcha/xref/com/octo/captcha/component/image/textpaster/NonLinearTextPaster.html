23   * Pastes characters in differents height lines in the background

28  public class NonLinearTextPaster extends AbstractTextPaster {
29

37          super(minAcceptedWordLength, maxAcceptedWordLength, colorGenerator);
38      }
39  
40      public NonLinearTextPaster(Integer minAcceptedWordLength, Integer maxAcceptedWordLength,
41                              ColorGenerator colorGenerator, Boolean manageColorPerGlyph) {
42          super(minAcceptedWordLength, maxAcceptedWordLength, colorGenerator, manageColorPerGlyph);
43      }
44  
45      /***
46       * Pastes the attributed string on the backround image and return the final image. Implementation must take into
47       * account the fact that the text must be readable by human and non by programs.
48       *
49       * @return the final image
50       *
51       * @throws com.octo.captcha.CaptchaException
52       *          if any exception accurs during paste routine.
53       */
54      public BufferedImage pasteText(final BufferedImage background,
55                                     final AttributedString attributedWord) throws CaptchaException {
56  
57          BufferedImage out = copyBackground(background);
58          Graphics2D g2 = pasteBackgroundAndSetTextColor(out, background);
59  
60          // convert string into a series of glyphs we can work with
61          ChangeableAttributedString newAttrString = new ChangeableAttributedString(g2,
62                  attributedWord, 2);
63  
64          // space out the glyphs with a little kerning
65          newAttrString.useMinimumSpacing(6);
66  
67          // shift string to a non-linear layout in the output image
68          newAttrString.shiftBoundariesToNonLinearLayout(background.getWidth(), background.getHeight());
69          
70          // now draw each glyph at the appropriate spot on the image.
71          if (isManageColorPerGlyph())
72              newAttrString.drawString(g2, getColorGenerator());
73          else
74              newAttrString.drawString(g2);
75  
76          g2.dispose();
77          return out;
78      }
79  }
