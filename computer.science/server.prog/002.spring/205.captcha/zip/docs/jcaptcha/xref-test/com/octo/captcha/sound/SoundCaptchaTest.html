54                  soundCaptcha.getChallenge() instanceof AudioInputStream);
55      }
56  
57      /***
58       * This test is for verifying if the audio captcha are different stream but have the same content.
59       */
60      public void testGetAudioChallenge() throws Exception {
61          Object challengeObject = soundCaptcha.getChallenge();
62          assertEquals(AudioInputStream.class, challengeObject.getClass());
63          AudioInputStream challengeAudioStream = (AudioInputStream) challengeObject;
64          AudioInputStream soundChallengeAudioStream = soundCaptcha.getSoundChallenge();
65          assertEquals(soundChallengeAudioStream.getFormat().toString(), challengeAudioStream.getFormat().toString());
66          assertEquals(soundChallengeAudioStream.getFrameLength(), challengeAudioStream.getFrameLength());
67      }
