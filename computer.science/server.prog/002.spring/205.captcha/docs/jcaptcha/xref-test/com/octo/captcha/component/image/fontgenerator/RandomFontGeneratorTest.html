49      public void setUp() {
50          this.randomFontGenerator =
51                  new RandomFontGenerator(new Integer(minFontSize), new Integer(maxFontSize));
52  
53          Font[] fontsList = new Font[2];
54          fontsList[0] = new Font("Courier", Font.BOLD, 10);
55          fontsList[1] = new Font("Arial", Font.BOLD, 10);
56  
57          this.randomFontGeneratorWithList =
58                  new RandomFontGenerator(new Integer(minFontSize), new Integer(maxFontSize), fontsList);
59      }

61      public void testGetFont() {
62          Font test = this.randomFontGenerator.getFont();
63          assertNotNull(test);
64      }
65  
66      public void testGetFontWithList() {
67          Font test = this.randomFontGeneratorWithList.getFont();
68          assertNotNull(test);
69          assertTrue(test.getName().startsWith("Arial"));
70      }

97      public void testGetFontWithBadFontPrefix() {
98  
99          this.randomFontGenerator.setBadFontNamePrefixes(new String[] {"Cour"});
100 
101         Font arial = new Font("Arial", Font.BOLD, 10);
102 
103         Font[] fontsList = new Font[2];
104         fontsList[0] = new Font("Courier", Font.BOLD, 10);
105         fontsList[1] = arial;
106 
107         java.util.List checkedFontList = this.randomFontGenerator.cleanFontList(fontsList);
108         assertEquals(1, checkedFontList.size());
109         assertEquals(arial, checkedFontList.get(0));
